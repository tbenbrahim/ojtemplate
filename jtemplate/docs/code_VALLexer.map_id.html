let map_id name=
    match name with
      "foreach" -> FOREACH
    | "in" -> IN
    | "while" -> WHILE
    | "function" -> FUNCTION
    | "if" -> IF
    | "else" -> ELSE
    | "template" -> TEMPLATE
    | "instructions" ->INSTRUCTIONS
    | "continue" -> CONTINUE
    | "break" -> BREAK
    | "return" -> RETURN
    | "for" -> FOR
    | "once" -> ONCE
    | "when" -> WHEN
    | "var" -> VAR
    | "let" -> VAR
    | "true" -> BOOLEAN(true)
    | "false" -> BOOLEAN(false)
    | "Void"  -> VOID
    | "import" -> IMPORT(false)
    | "use" -> IMPORT(true)
    | "switch" -> SWITCH
    | "case" -> CASE
    | "default" -> DEFAULT
    | "try" -> TRY
    | "catch" -> CATCH
    | "finally" -> FINALLY
    | "throw" -> THROW
    | _ ->  ID(name)
